def rule_rating(row):
    if row["cover"] == 'Твёрдый переплёт':
        return 1
    elif row["cover"] == 'Мягкий заламинированный картон':
        return 2
    elif row["cover"] == 'Плотный картон':
        return 3
    else:
        return 0
